def predict_crop(N, P, K, temperature, humidity, ph, rainfall):
    input_data = np.array([[N, P, K, temperature, humidity, ph, rainfall]])
    prediction = rf_model.predict(input_data)
    return le.inverse_transform(prediction)[0]
